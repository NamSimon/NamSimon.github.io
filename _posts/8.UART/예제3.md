#include <avr/io.h>
#include <avr/interrupt.h>

void init();
void txd(char ch);
void txd_string(char *str);

volatile unsigned char Cmd_U0 = 0x30;

int main(void){
	unsigned char seg[16] = {0x3f, 0x06, 0x5b, 0x4f, 0x66,0x6d,0x7d,0x07,
	0x7f, 0x67, 0x77, 0x7c, 0x58,0x5e,0x79, 0x71};
	unsigned char ch;
	init();

	txd_string("\n\r Example 10-2\n\r");
	txd_string("\r Enter Key : (0~9)\n\r");
	PORTG = 0x08;
	while(1) {
		ch = Cmd_U0;
		PORTC = seg[ch-0x30];
	}
}

void txd(char ch)
{
	while(!(UCSR0A & 0x20)); // UDRE0=1 ?(전송데이터 empty?)
	UDR0 = ch;
} /* 1 byte data를 전송 */

void txd_string(char *str)
{
	int i=0;
	while(1) {
		if (str[i] == '\0') break;
		txd(str[i++]);
	}
}
	


void init()
{
	DDRA = 0xFF;
	DDRB = 0x10;
	DDRC = 0xFF;
	DDRG = 0x0F;
	DDRE = 0x02;
	// 나머지 모두 comment 처리
	UCSR0A = 0x00;
	UCSR0B = 0x98;
	UCSR0C = 0x06;
	UBRR0H = 0;
	UBRR0L = 0x67;
	sei();
}

ISR(USART0_RX_vect) // keyboard input 시 interrupt 발생
{
	Cmd_U0 = UDR0; /* Cmd_U0 : global 변수 */
}
